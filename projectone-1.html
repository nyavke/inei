# Подсчёт поднятых пальцев
def recognize_gesture(landmarks):
    def is_up(tip, pip):
        return landmarks[tip].y < landmarks[pip].y
    fingers = [is_up(8,6), is_up(12,10), 
               is_up(16,14), is_up(20,18)]
    cnt = sum(fingers)
    if cnt == 4: return "stop"
    if cnt == 0: return "back"
    if cnt == 1 and fingers[0]: return "forward"
    if cnt == 2 and fingers[0] and fingers[1]: return "right"
    if cnt == 3: return "left"
    # большой палец вверх → like (сжать)
    if cnt == 0 and thumb_up(landmarks): return "like"
    # кольцо большой+указательный → ok (разжать)
    if is_ok_sign(landmarks): return "ok"
    return "none"

# Стабилизация и отправка в MQTT
client = mqtt.Client()
client.connect("broker.emqx.io", 1883)
history = []
while True:
    gesture = recognize_gesture(landmarks)
    history.append(gesture)
    if len(history) > 5: history.pop(0)
    stable = max(set(history), key=history.count)
    if stable != "none" and stable != last_cmd:
        client.publish("ligarobotov/darkfaerie", stable)
        last_cmd = stable

— Связь с LEGO Mindstorms

→ MQTT

Робот подписан на топик

На LEGO EV3 запущен скрипт на MicroPython, который слушает топик ligarobotov/darkfaerie и преобразует команды в движения моторов.

→ forward

Движение вперёд

Оба мотора вращаются вперёд. Робот едет прямо.

→ back

Движение назад

Оба мотора вращаются назад.

→ left / right

Повороты на месте

Один мотор вращается вперёд, другой назад — робот разворачивается.

→ stop

Немедленная остановка

Моторы блокируются. Безопасное торможение.

→ like

Сжатие захвата

Мотор захвата вращается до упора — клешня смыкается и удерживает объект.

→ ok

Разжатие захвата

Мотор захвата вращается в обратную сторону — клешня раскрывается и отпускает объект.
